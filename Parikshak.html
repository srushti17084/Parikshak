Implement Linear Queue Using ADT Array

#include <stdio.h> #include <stdlib.h> #define MAX 100 // Queue structure using an array int queue[MAX]; int front = -1; int rear = -1; // Function to add an element to the queue (Enqueue) void enqueue(int value) { if (rear == MAX - 1) { printf("-1\n"); } else { if (front == -1) { front = 0; // Initialize front for first element } rear++; queue[rear] = value; } } // Function to remove an element from the queue (Dequeue) void dequeue() { if (front == -1 || front > rear) { printf("-1\n"); } else { front++; } } // Function to display the elements in the queue void display() { if (front == -1 || front > rear) { printf("-1\n"); } else { for (int i = front; i <= rear; i++) { printf("%d ", queue[i]); } printf("\n"); } } // Main function to run the menu-driven program int main() { int choice, value; do { scanf("%d", &choice); switch (choice) { case 1: scanf("%d", &value); enqueue(value); break; case 2: dequeue(); break; case 3: display(); break; case 4: break; default: printf("-1\n"); } } while (choice != 4); return 0; }
